f93_system_has_ancient_relay = {
	OR = {
		has_megastructure = dandelion_relay
		has_megastructure = dandelion_relay_star
	}
}

safe_highway_start_end = {
	NOR = {
		exists = owner
		any_fleet_in_system = { exists = this }
		has_star_flag = chosen_cluster
		has_star_flag = empire_cluster
		# any_neighbor_system = { exists = owner }
		closest_system = {
			limit = { exists = owner }
			max_steps = 3
		}
	}
}

	is_same_value = event_target:galactic_union_state

is_shrouded_ai = {
	is_ai = yes
	is_country_type = default
	NOR = {
		has_country_flag = chosen_empire
	}
	capital_scope = {
		NOT = {
			closest_system = {
				limit = { f93_system_has_ancient_relay = yes }
				max_steps = 20
			}
		}
	}
}
is_shrouded_player = {
	is_ai = no
	capital_scope = {
		NOT = {
			closest_system = {
				limit = { f93_system_has_ancient_relay = yes }
				max_steps = 20
			}
		}
	}
}


f93_is_close_to_highway = {
	OR = {
		any_system_within_border = { f93_system_has_ancient_relay = yes }
		any_system_within_border = {
			any_neighbor_system = { f93_system_has_ancient_relay = yes }
		}
	}
}

f93_should_not_be_destroyed = {
	OR = {
		any_system_planet = {
			OR = {
				is_ringworld = yes
				has_modifier = holy_planet
				#If you only have a planet that should be protected add it here and the entire system will be safe from me.

			}
		}
		AND = {
			exists = space_owner
			space_owner = { is_same_value = event_target:f93_ralis_country }
		}
		is_star_class = sc_black_hole
		has_star_flag = star_cracked # already destroyed
		has_star_flag = f93_empty_system_on_galcrossing #have special star class "no Star"
		has_star_flag = crisis_sapwn_exclude #generally marks special systems

		#Exlcluding a bunch of mods, feel free to add your own or request me to add it.

		has_star_flag = eaw_system
		has_star_flag = giga_system_four
		has_star_flag = giga_system_three

	}
}

f93_country_is_connected_to_gal_crossing = {
	any_system_within_border = {
		OR = {
			AND = {
				has_megastructure = hyper_relay
				any_neighbor_system = {
					f93_system_has_ancient_relay = yes
				}
			}
			f93_system_has_ancient_relay = yes
		}
		is_system_connected_to_relay_network = yes
	}
}
