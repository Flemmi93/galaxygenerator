#Corvette
f93_create_corvettes = {
	while = {
		count = 8
		limit = { owner = { has_resource = { type = alloys amount < 1500 } } }
		owner = { add_resource = { alloys = -100 } }
		create_fleet = {
			settings = {
				can_upgrade = yes
				can_change_composition = yes
				can_change_leader = yes	
			}
			effect = {
				create_ship = {
					name = random
					random_existing_design = corvette
				}
			}
		}
	}
}

#Corvette Fleet
f93_create_corvette_fleet = {
	create_fleet =  {
		settings = {
			can_upgrade = yes
			can_change_composition = no
			can_change_leader = yes	
		}
		effect = {
			while = {
				count = root.f93_fleet_size
				create_ship = {
					name = random
					random_existing_design = corvette
				}
			}
		}
	}
}

#Cruiser Fleet
f93_create_cruiser_fleet = {
	create_fleet =  {
		settings = {
			can_upgrade = yes
			can_change_composition = no
			can_change_leader = yes	
		}
		effect = {
			set_owner = root
			while = {
				count = root.f93_fleet_size
				while = {
					count = 2
					create_ship = {
						name = random
						random_existing_design = corvette
					}
				}
				while = {
					count = 2
					create_ship = {
						name = random
						random_existing_design = frigate
					}
				}
				while = {
					count = 2
					create_ship = {
						name = random
						random_existing_design = destroyer
					}
				}
				while = {
					count = 1
					create_ship = {
						name = random
						random_existing_design = cruiser
					}
				}
			}
		}
	}
}
